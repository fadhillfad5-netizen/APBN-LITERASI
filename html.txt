    // INITIALIZATION ON DOM READY
    window.addEventListener('DOMContentLoaded', () => {
      // Create Lucide Icons
      lucide.createIcons();
      
      // Load interactive breakdown cards
      buildSectorGrid();

      // Mask and format inputs
      const txtInput = document.getElementById('pajakInput');
      txtInput.addEventListener('input', function() {
        let numericValue = cleanNumeric(this.value);
        if (numericValue === 0) {
          this.value = '';
        } else {
          this.value = numericValue.toLocaleString('id-ID');
        }
        currentNominal = numericValue;
        triggerDataUpdate();
      });

      document.getElementById('btnCek').addEventListener('click', () => {
        let numericValue = cleanNumeric(txtInput.value);
        if (numericValue > 0) {
          currentNominal = numericValue;
          triggerDataUpdate();
        }
      });

      // Build initial Charts
      initDonutChart();
      initRealityChart();

      // Autoselect first item (Pendidikan)
      selectSector('pendidikan');
    });

    // Helper functions
    function cleanNumeric(str) {
      return parseInt(str.replace(/\D/g, '')) || 0;
    }

    // Dynamic scale helper that switches calculations based on Simulator mode
    function getSectorValue(sector) {
      if (isSdaModeActive) {
        return sector.tragicValue;
      }
      return currentNominal * (sector.percentage / 100);
    }

    function formatRupiah(num) {
      if (num < 10) {
        return 'Rp' + num.toLocaleString('id-ID', { minimumFractionDigits: 0, maximumFractionDigits: 2 });
      }
      return 'Rp' + Math.round(num).toLocaleString('id-ID');
    }

    function setNominal(val) {
      currentNominal = val;
      document.getElementById('pajakInput').value = val.toLocaleString('id-ID');
      triggerDataUpdate();
    }

    // Build static selection cards
    function buildSectorGrid() {
      const container = document.getElementById('sectorsContainer');
      container.innerHTML = '';

      sectorsList.forEach(sector => {
        const valueShare = getSectorValue(sector);
        const cardBtn = document.createElement('button');
        cardBtn.id = `card-${sector.id}`;
        cardBtn.className = "flex items-center gap-3.5 p-4 bg-white hover:bg-slate-50 border border-slate-100 rounded-2xl text-left transition-all hover:-translate-y-0.5 active:scale-95 focus:outline-none focus:ring-2 focus:ring-slate-300 w-full shadow-sm";
        cardBtn.onclick = () => selectSector(sector.id);

        cardBtn.innerHTML = `
          <div class="p-3 rounded-xl flex-shrink-0" style="background-color: ${sector.lightBg}; color: ${sector.color}">
            <i data-lucide="${sector.icon}" class="w-5 h-5"></i>
          </div>
          <div class="flex-1 min-w-0">
            <div class="flex justify-between items-center gap-1">
              <span class="text-xs font-bold text-slate-800 truncate">${sector.name}</span>
              <span class="text-[10px] font-extrabold px-1.5 py-0.5 rounded-full" id="percent-${sector.id}" style="background-color: ${sector.lightBg}; color: ${sector.color}">${sector.percentage}%</span>
            </div>
            <span class="text-sm font-black text-slate-700 mt-1 block transition-all duration-300" id="cost-${sector.id}">
              ${formatRupiah(valueShare)}
            </span>
          </div>
        `;
        container.appendChild(cardBtn);
      });
      lucide.createIcons();
    }

    // Interactive event when sector gets active
    function selectSector(sectorId) {
      const sector = sectorsList.find(s => s.id === sectorId);
      if (!sector) return;

      // Update Active State on UI Buttons
      sectorsList.forEach(s => {
        const btn = document.getElementById(`card-${s.id}`);
        if (s.id === sectorId) {
          btn.classList.remove('border-slate-100', 'bg-white');
          btn.classList.add('border-slate-950', 'bg-slate-50', 'ring-4', 'ring-slate-900/5');
        } else {
          btn.classList.add('border-slate-100', 'bg-white');
          btn.classList.remove('border-slate-950', 'bg-slate-50', 'ring-4', 'ring-slate-900/5');
        }
      });

      // Update Detail Banner Title, Badges and Micro-copy
      const valueShare = getSectorValue(sector);
      document.getElementById('activeSectorBadge').innerText = isSdaModeActive ? `KRISIS: ${sector.name} (KOLAPS)` : `SEKTOR: ${sector.name} (${sector.percentage}%)`;
      document.getElementById('activeSectorTitle').innerText = sector.name;
      document.getElementById('activeSectorDesc').innerText = isSdaModeActive ? sector.tragicDesc : sector.desc;
      document.getElementById('activeSectorValue').innerText = formatRupiah(valueShare);

      // Highlight Slice inside Chart.js
      if (donutChartInstance) {
        const index = sectorsList.findIndex(s => s.id === sectorId);
        donutChartInstance.setActiveElements([{ datasetIndex: 0, index: index }]);
        donutChartInstance.tooltip.setActiveElements([{ datasetIndex: 0, index: index }], { x: 0, y: 0 });
        donutChartInstance.update();
      }
    }

    // Trigger update on state change (nominal change or toggle simulator)
    function triggerDataUpdate() {
      // Update Center Label inside Donut
      document.getElementById('centerTotalLabel').innerText = isSdaModeActive ? "Anjlok!" : formatRupiah(currentNominal);

      // Update cost elements on Cards dynamically to match current simulation state
      sectorsList.forEach(sector => {
        const valueShare = getSectorValue(sector);
        const costEl = document.getElementById(`cost-${sector.id}`);
        if (costEl) costEl.innerText = formatRupiah(valueShare);

        // Change badge label to warn collapse if SDA mode is active
        const badgeEl = document.getElementById(`percent-${sector.id}`);
        if (badgeEl) {
          if (isSdaModeActive) {
            badgeEl.innerText = "Kolaps";
            badgeEl.style.backgroundColor = "#fee2e2";
            badgeEl.style.color = "#ef4444";
          } else {
            badgeEl.innerText = `${sector.percentage}%`;
            badgeEl.style.backgroundColor = sector.lightBg;
            badgeEl.style.color = sector.color;
          }
        }
      });

      // Update active selected banner contents to match the current selected sector calculations
      const activeCard = document.querySelector('[id^="card-"].border-slate-950');
      if (activeCard) {
        const activeId = activeCard.id.replace('card-', '');
        const sector = sectorsList.find(s => s.id === activeId);
        if (sector) {
          const valueShare = getSectorValue(sector);
          document.getElementById('activeSectorBadge').innerText = isSdaModeActive ? `KRISIS: ${sector.name} (KOLAPS)` : `SEKTOR: ${sector.name} (${sector.percentage}%)`;
          document.getElementById('activeSectorDesc').innerText = isSdaModeActive ? sector.tragicDesc : sector.desc;
          document.getElementById('activeSectorValue').innerText = formatRupiah(valueShare);
        }
      }

      // Refresh donut dataset with animations
      if (donutChartInstance) {
        let newDataset = [];
        if (isSdaModeActive) {
          // Collapse values representing an economic shock
          newDataset = sectorsList.map(s => s.tragicValue); 
        } else {
          // Healthy normal values
          newDataset = sectorsList.map(s => currentNominal * (s.percentage / 100));
        }
        donutChartInstance.data.datasets[0].data = newDataset;
        donutChartInstance.update();
      }
    }

    // INITIALIZE CHART.JS: Donut
    function initDonutChart() {
      const ctx = document.getElementById('pajakDonutChart').getContext('2d');
      const dataset = sectorsList.map(s => getSectorValue(s));
      const colors = sectorsList.map(s => s.color);
      const labels = sectorsList.map(s => s.name);

      donutChartInstance = new Chart(ctx, {
        type: 'doughnut',
        data: {
          labels: labels,
          datasets: [{
            data: dataset,
            backgroundColor: colors,
            hoverOffset: 12,
            borderWidth: 4,
            borderColor: '#ffffff',
            borderRadius: 8
          }]
        },
        options: {
          responsive: true,
          maintainAspectRatio: false,
          plugins: {
            legend: { display: false },
            tooltip: {
              callbacks: {
                label: function(context) {
                  let val = context.raw;
                  return ` ${context.label}: ${formatRupiah(val)}`;
                }
              }
            }
          },
          onClick: (evt, elements) => {
            if (elements.length > 0) {
              const idx = elements[0].index;
              const sectorId = sectorsList[idx].id;
              selectSector(sectorId);
            }
          },
          cutout: '74%'
        }
      });
    }

    // TOGGLE MODE SIMULATOR (Pajak Nyata vs SDA Only)
    function setSimulatorMode(mode) {
      isSdaModeActive = (mode === 'sda');
      
      const btnReal = document.getElementById('btnRealStructure');
      const btnSda = document.getElementById('btnSdaOnly');
      const stateNormal = document.getElementById('stateNormal');
      const stateUnstable = document.getElementById('stateUnstable');
      const chartCard = document.getElementById('chartCard');
      const chartTitle = document.getElementById('chartTitle');
      const centerLabelTop = document.getElementById('centerLabelTop');
      const activeLabelBottom = document.getElementById('activeLabelBottom');

      if (isSdaModeActive) {
        // PLAY DRAMATIC WARNING SOUND
        playWarningSound();

        // Switch buttons layout
        btnSda.className = "flex-1 md:flex-none px-5 py-3 rounded-xl font-bold text-xs md:text-sm transition-all flex items-center justify-center gap-2 bg-orange-500 text-white shadow-md shadow-orange-500/30";
        btnReal.className = "flex-1 md:flex-none px-5 py-3 rounded-xl font-bold text-xs md:text-sm transition-all flex items-center justify-center gap-2 text-slate-500 hover:text-slate-800";
        
        // Show unstable layout info
        stateNormal.classList.add('hidden');
        stateUnstable.classList.remove('hidden');

        // Apply shake animation class to chartcard
        chartCard.classList.add('unstable-wiggle', 'border-red-500/20');
        chartTitle.innerHTML = `<span class="text-red-500 font-extrabold flex items-center justify-center gap-1"><i data-lucide="alert-triangle" class="w-4 h-4 animate-bounce"></i> Anggaran Kolaps</span>`;
        centerLabelTop.innerText = "KEUANGAN";
        activeLabelBottom.innerText = "Hasil Patungan (Kolaps):";

        // Instantly force-select current active sector to refresh card texts & tragic values
        const activeCard = document.querySelector('[id^="card-"].border-slate-950');
        const activeId = activeCard ? activeCard.id.replace('card-', '') : 'pendidikan';
        selectSector(activeId);
      } else {
        // PLAY HAPPY SUCCESS CHIME SOUND
        playSuccessSound();

        // Reset to normal structured state
        btnReal.className = "flex-1 md:flex-none px-5 py-3 rounded-xl font-bold text-xs md:text-sm transition-all flex items-center justify-center gap-2 bg-white text-slate-800 shadow-sm";
        btnSda.className = "flex-1 md:flex-none px-5 py-3 rounded-xl font-bold text-xs md:text-sm transition-all flex items-center justify-center gap-2 text-slate-500 hover:text-slate-800";

        stateNormal.classList.remove('hidden');
        stateUnstable.classList.add('hidden');

        // Clean shake animations
        chartCard.classList.remove('unstable-wiggle', 'border-red-500/20');
        chartTitle.innerText = "Kue Distribusi Alokasi";
        centerLabelTop.innerText = "Porsi Kamu";
        activeLabelBottom.innerText = "Hasil Patunganmu:";

        // Force select to refresh normal text
        const activeCard = document.querySelector('[id^="card-"].border-slate-950');
        const activeId = activeCard ? activeCard.id.replace('card-', '') : 'pendidikan';
        selectSector(activeId);
      }

      triggerDataUpdate();
      lucide.createIcons();
    }
